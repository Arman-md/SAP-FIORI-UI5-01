``` html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="author" content="arman">
    <meta name="keywords" content="learn html, html training">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Basics of HTML 5</title>
</head>

<body>
    <!-- heading tags -->
    <h1>Welcome to HTML5</h1>
    <h2>Welcome to HTML5</h2>
    <h3>Welcome to HTML5</h3>
    <h4>Welcome to HTML5</h4>
    <h5>Welcome to HTML5</h5>
    <h6>Welcome to HTML5</h6>

    <!-- apargraph and bold and emphasis tag -->
    <p>HTML Stands for <em> <b>hyper text markup language</b> </em> and it is used to create static web pages</p>

    <!-- anchor tag -->
    <a href="https://www.youtube.com/">click here for youtube</a>

    <!-- line break tag (for new line) -->
    <br>
    <!-- image tag -->
    <!-- Image tag in the anchor to create a hotspot on image -->
    <a href="https://www.lamborghini.com/en-en/models/urus/urus-se#val-ht">
        <img src="images/blue-lamborghini-urus-suv-3i.jpg" width="30%" height="30%">
    </a>

    <!-- Usertname and password input from user -->
     <!-- id will be always unique to each html element -->
    <br>
    <label>user name</label>
    <input name="user" id="iduser">
    <br>
    <label>password</label>
    <input name="password" id="idpwd" type="password">

    <br>
    <!-- email accenpting and number 1-10 only accepting fields and submit button-->
    <input name="email" type="email">
    <input name="number" type="number" min="1" max="10">

    <input name="submit" type="submit" >

</body>

</html>

```
